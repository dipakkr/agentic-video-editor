   - class TrackMeta(BaseModel): track_id: str, title: str = "", artist: str = "",
     genre: str = "", bpm: float, duration_s: float, energy_curve: list[float] = [],
     downbeat_offset_s: float = 0.0, license: str = "", source: str = "", file: str = ""
   - def load_library(dir: Path) -> list[TrackMeta]
     Reads all *.json in dir that contain both "track_id" and "bpm" keys (skip others,

3. backend/ave/music/grid.py
   - def grid_for_track(meta: TrackMeta, audio_path: str | None = None) -> BeatGrid
     If audio_path given, try ave.beat.snap.grid_from_librosa(audio_path); on None/exception
     fall back to synthetic_grid(meta.bpm, meta.duration_s) with beats shifted by
     meta.downbeat_offset_s (all beats + downbeats shifted, non-negative times only).
   - def drop_time(meta: TrackMeta) -> float
     Time of the energy_curve maximum: index_of_max/len(curve) * duration_s. Empty
     curve -> 0.0. First occurrence wins on ties (determinism).

4. backend/ave/agents/music.py
   - def apply_music(edl: EDL, manifests: list[ClipManifest], library_dir: Path,
                     settings: Settings | None = None) -> EDL

        unchanged (no bump) — graceful no-music path.
     b) grid = grid_for_track(track, audio_path=str(library_dir / track.file) if that
        file exists else None)
     c) snapped = snap_edl(edl, grid, tolerance_ms=settings.ave_beat_snap_tolerance_ms)

     d) Align the drop: find the manifest Highlight with the highest score whose source
        window overlaps a timeline segment of that clip; compute its timeline position
        (edl.timeline_offset_of(segment) + (highlight.start_s - segment.in_)/segment.speed,
        clamped into the segment); set music.offset_s = max(0.0, drop_time(track) -
        that_timeline_position). No highlights -> offset_s stays 0.0.
     e) Set music.track_id, ducking=True, duck_db from existing default, fade_in_s=0.5,
        fade_out_s=1.5. Return the new EDL.
     Include a module docstring explaining the pass. Never raise for missing optional
     pieces; the EDL must always come back valid.

   - class Cue(BaseModel): start_s: float, end_s: float, text: str,
     words: list[CueWord] = []
     All times are TIMELINE times (not source-clip times).
   - def build_cues(edl: EDL, manifests: list[ClipManifest],
                    mode: str = "sentence") -> list[Cue]
     For each timeline segment (in order): find its manifest by seg.source_clip; take
     transcript Words whose source midpoint lies within [seg.in_, seg.out]; remap each
     word to timeline time: offset = edl.timeline_offset_of(seg.id);
     t = offset + (word.start_s - seg.in_)/seg.speed (same for end), clamped into
     [offset, offset + seg.timeline_duration_s]. Skip words flagged is_filler.
     Grouping: mode "sentence" -> one Cue per source TranscriptSegment slice (text = the
     words joined); mode "phrase" -> greedy groups of <= 5 words AND <= 2.2s span.
     Cues sorted by start_s; empty transcript -> [].

     karaoke_bold/phrase_pop -> "phrase"; clean_subtitle -> "sentence"; none -> "sentence".

   - def to_ass(cues: list[Cue], captions: Captions, output: OutputSpec) -> str

     captions.font/font_size overrides; MarginV computed from captions.position_y and a

     height (platform UI overlap), for 16:9 minimum 5%.
     [Events]: for karaoke_bold, each Dialogue line is one phrase cue whose text uses
     per-word karaoke tags {\kNN}word (NN = word duration in centiseconds, >=1);
     for phrase_pop/clean_subtitle, plain text lines. Dialogue times H:MM:SS.cc.
   - def format_srt_time(t: float) -> str and def format_ass_time(t: float) -> str
